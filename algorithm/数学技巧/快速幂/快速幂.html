public:
    // é¢˜ç›®æè¿°:

    double Power(double base, int exp) {
        double res = 1;
        
        // å¤„ç†expä¸ºè´Ÿçš„æƒ…å†µ
        int pow = exp < 0 ? -1 * exp : exp;
        while (pow != 0) {
            if ((pow & 1) == 1) {
                res *= base;
            }
            pow >>= 1;
            base *= base;
        }
        return exp < 0 ? 1.0 / res : res;
    }
};
